  ImageMap:
    HVM64: ami-01aab85a5e4a5a0fe
    # ...

Resources:
  EC2Instance:
    Type: "AWS::EC2::Instance"
    Properties:
      ImageId: !FindInMap [ImageMap, HVM64]
      InstanceType: t2.micro

Outputs:
   InstanceID:
    Description: The Instance ID
    Value: !Ref EC2Instance
